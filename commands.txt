git init # initalizes empty git repo

git status # info about current state of git repo

git checkout -b <branch name>. # switching to the new branch between branches 

git checkout <branch name> - switching between branches

git add <file name> use to stage the unstaged files 

git commit -m "<message>"  - used to track the staged file 

git restore --staged <filename:>    - to untrack the staged file 

git fetch - to add the different branches to the local 

git pull origin dev - to pull the files from dev branch to origin 

git push origin master - to push the changes from origin to master 

git pull origin dev --rebase

git remote add origin https://github.com/vivekubale09/github-for-devops-workshop.git - to link your local Git repository with a remote repository on GitHub.
git remote -v - to check the origin link
